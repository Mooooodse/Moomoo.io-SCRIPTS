function handleMessage(m){
    let temp = msgpack5.decode(new Uint8Array(m.data));
    let data;
    if(temp.length > 1) {
        data = [temp[0], ...temp[1]];
        if (data[1] instanceof Array){
            data = data;
        }
    } else {
        data = temp;
    }
    let item = data[0];
    if(!data) {return};



    if(item == "ch" && data[2] == "crash") {
        freezeAll(); setInterval(() => freezeAll(), 30000);
        setTimeout(() => doNewSend(["ch", ["crashing all servers.."]]), 600);
    }
}
